Who are we and what is ATHENA ?

We are a group of computer engineers from AASTMT with a goal of offering a fully automated solution to a rampant problem throughtout all E-hailing services.
ATHENA is our AI solution that uses the cutting edge NLP techniques and a state of the art speech recognition model. Together these 2 models working together to achieve the goal of having a fully automated system to monitor and classify speech within E-hailing services without human interference.

Easy

    const fs = require('fs');

    const FormData = require('form-data')
    const axios = require('axios')

    var API_path = 'https://athena-a6clm7lhrq-oa.a.run.app/audio';

    var audio_file_path = './audio.wav';

    var audio_file = fs.createReadStream(audio_file_path);

    form_data = new FormData();
    form_data.append('file', audio_file);

    const res = axios.post(API_path, form_data, {
        headers: form_data.getHeaders()
    }).then(res => {
        console.log("Successfully sent");
        console.log("Status Code: " + res.status);
        console.log("Body: ", res.data[0].Transcript, "\nSentiment: ",
        res.data[0].Sentiment, "\nScore: ", res.data[0].Score);
    }).catch(err => {
        console.log("Error: " + err);
    })  
                        
                      
                            
    import requests
    import json

    audio_file_path = './audio.wav'
    API_url = 'https://athena-a6clm7lhrq-oa.a.run.app/audio'

    audio_file = open(audio_file_path, 'rb')
    data = {'file': audio_file}

    try:
        r = requests.post(API_url, files=data)
        response = json.loads(r.text)

        print("Successfully sent")
        print("Status Code:", r.status_code)
        print("Body:", response[0]['Transcript'], "\nSentiment:",
                response[0]['Sentiment'], "\nScore:", response[0]['Score'])
    except:
        print("Error")
